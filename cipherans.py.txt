def encrypt_cypher(plaintext,n):
    cipher_ans = ""
    for x in range(len(plaintext)):
        char = plaintext[x]
        
        #if space is present then add space
        if char==" " or char == ',' or char == '.' or char == ';':
            cipher_ans+=" "
        # if a character is uppercase then encrypt it 
        elif (char.isupper()):
            cipher_ans += chr((ord(char) + n-65) % 26 + 65)
        # if a character is lowercase then encrypt it 
        
        else:
            cipher_ans += chr((ord(char) + n-97) % 26 + 97)

    return  cipher_ans
